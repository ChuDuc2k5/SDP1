BEGIN;

CREATE EXTENSION IF NOT EXISTS "pgcrypto";

DROP TABLE IF EXISTS rates CASCADE;
DROP TABLE IF EXISTS images CASCADE;
DROP TABLE IF EXISTS promotions CASCADE;
DROP TABLE IF EXISTS booking_policies CASCADE;
DROP TABLE IF EXISTS otps CASCADE;
DROP TABLE IF EXISTS bookings CASCADE;
DROP TABLE IF EXISTS settings CASCADE;
DROP TABLE IF EXISTS cabins CASCADE;
DROP TABLE IF EXISTS users CASCADE;

DROP TYPE IF EXISTS role_enum CASCADE;
DROP TYPE IF EXISTS booking_status_enum CASCADE;

CREATE TABLE users (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "fullName" VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password TEXT NOT NULL,
    phone VARCHAR(30),
    "nationalId" VARCHAR(50),
    "dateOfBirth" DATE,
    gender VARCHAR(20),
    address TEXT,
    nationality VARCHAR(100),
    role VARCHAR(30) NOT NULL DEFAULT 'user',
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    CHECK (role IN ('user', 'customer', 'cabinOwner', 'admin'))
);

CREATE TABLE cabins (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(255) NOT NULL,
    "maxCapacity" INT NOT NULL,
    "regularPrice" NUMERIC(12,2) NOT NULL,
    discount NUMERIC(12,2) DEFAULT 0,
    image TEXT NOT NULL,
    description TEXT,
    location TEXT,
    amenities TEXT,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    CHECK ("maxCapacity" > 0),
    CHECK ("regularPrice" > 0),
    CHECK (discount >= 0)
);

CREATE TABLE booking_policies (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "cabinId" UUID NOT NULL REFERENCES cabins(_id) ON DELETE CASCADE,
    "breakfastPrice" NUMERIC(12,2) DEFAULT 15,
    "miniBookingLength" INT DEFAULT 1,
    "maxBookingLength" INT DEFAULT 30,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    UNIQUE ("cabinId"),
    CHECK ("breakfastPrice" >= 0),
    CHECK ("miniBookingLength" > 0),
    CHECK ("maxBookingLength" >= "miniBookingLength")
);

CREATE TABLE promotions (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "cabinId" UUID NOT NULL REFERENCES cabins(_id) ON DELETE CASCADE,
    "discountPercent" NUMERIC(5,2) NOT NULL,
    "startDate" DATE NOT NULL,
    "endDate" DATE NOT NULL,
    "isActive" BOOLEAN DEFAULT TRUE,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    CHECK ("endDate" >= "startDate"),
    CHECK ("discountPercent" >= 0 AND "discountPercent" <= 100)
);

CREATE TABLE images (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "cabinId" UUID NOT NULL REFERENCES cabins(_id) ON DELETE CASCADE,
    "imageUrl" TEXT NOT NULL,
    "name" VARCHAR(255),
    "isCover" BOOLEAN DEFAULT FALSE,
    "createdAt" TIMESTAMP DEFAULT NOW()
);

CREATE TABLE bookings (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "userId" UUID NOT NULL REFERENCES users(_id) ON DELETE CASCADE,
    "cabinId" UUID NOT NULL REFERENCES cabins(_id) ON DELETE CASCADE,
    "startDate" DATE NOT NULL,
    "endDate" DATE NOT NULL,
    "numNights" INT NOT NULL DEFAULT 1,
    "numGuests" INT NOT NULL DEFAULT 1,
    "cabinPrice" NUMERIC(12,2) NOT NULL,
    "extrasPrice" NUMERIC(12,2) DEFAULT 0,
    "totalPrice" NUMERIC(12,2) NOT NULL DEFAULT 0,
    status VARCHAR(30) NOT NULL DEFAULT 'unconfirmed',
    "hasBreakfast" BOOLEAN DEFAULT FALSE,
    "isPaid" BOOLEAN DEFAULT FALSE,
    observations TEXT,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    CHECK ("endDate" > "startDate"),
    CHECK ("numNights" > 0),
    CHECK ("numGuests" > 0),
    CHECK (status IN ('unconfirmed', 'confirmed', 'checked-in', 'checked-out', 'cancelled', 'pending'))
);

CREATE TABLE rates (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "userId" UUID NOT NULL REFERENCES users(_id) ON DELETE CASCADE,
    "cabinId" UUID NOT NULL REFERENCES cabins(_id) ON DELETE CASCADE,
    "bookingId" UUID REFERENCES bookings(_id) ON DELETE SET NULL,
    rating INT NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    UNIQUE ("bookingId")
);

CREATE TABLE otps (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) NOT NULL,
    otp VARCHAR(10) NOT NULL,
    "expiresAt" TIMESTAMP NOT NULL,
    "userId" UUID REFERENCES users(_id) ON DELETE CASCADE,
    "createdAt" TIMESTAMP DEFAULT NOW()
);

CREATE TABLE settings (
    _id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "miniBookingLength" INT NOT NULL DEFAULT 1,
    "maxBookingLength" INT NOT NULL DEFAULT 30,
    "maxNumberOfGuests" INT NOT NULL DEFAULT 10,
    "breakfastPrice" NUMERIC(12,2) NOT NULL DEFAULT 15,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW(),
    CHECK ("miniBookingLength" > 0),
    CHECK ("maxBookingLength" >= "miniBookingLength"),
    CHECK ("maxNumberOfGuests" > 0),
    CHECK ("breakfastPrice" >= 0)
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_bookings_user_id ON bookings("userId");
CREATE INDEX idx_bookings_cabin_id ON bookings("cabinId");
CREATE INDEX idx_bookings_status ON bookings(status);
CREATE INDEX idx_bookings_start_date ON bookings("startDate");
CREATE INDEX idx_bookings_end_date ON bookings("endDate");
CREATE INDEX idx_rates_user_id ON rates("userId");
CREATE INDEX idx_rates_cabin_id ON rates("cabinId");
CREATE INDEX idx_rates_booking_id ON rates("bookingId");
CREATE INDEX idx_otps_email ON otps(email);
CREATE INDEX idx_otps_expires_at ON otps("expiresAt");
CREATE INDEX idx_images_cabin_id ON images("cabinId");
CREATE INDEX idx_promotions_cabin_id ON promotions("cabinId");
CREATE INDEX idx_booking_policies_cabin_id ON booking_policies("cabinId");

    b._id AS "bookingId",
    u."fullName",
    u.email,
    c."name" AS "cabinName",
    b."startDate",
    b."endDate",

    b."numGuests",
    b.status,
    b."totalPrice"
FROM bookings AS b
JOIN users AS u ON b."userId" = u._id
JOIN cabins AS c ON b."cabinId" = c._id
ORDER BY b."startDate";

FROM bookings AS b
